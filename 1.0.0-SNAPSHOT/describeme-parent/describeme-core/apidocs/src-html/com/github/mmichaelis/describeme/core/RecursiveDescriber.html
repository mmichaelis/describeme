017package com.github.mmichaelis.describeme.core;
018
019import org.jetbrains.annotations.Contract;
020import org.jetbrains.annotations.NotNull;
021import org.jetbrains.annotations.Nullable;
022
023import java.util.function.BiConsumer;
024
025/**
026 * Extension to {@link Describer} which handles recursion.
027 *
028 * @since 1.0.0
029 */
030public interface RecursiveDescriber extends Describer {
031
032  /**
033   * {@inheritDoc}
034   * <p>
035   * Default implementation with depth {@link DescriberProperties#MAX_DEPTH} which is configurable
036   * through system properties.
037   * </p>
038   * <p>
039   * If your value can be converted to a stream it is recommended to extend
040   * {@link AbstractStreamDescriber} which will take care of looping through the
041   * stream.
042   * </p>
043   *
044   * @since 1.0.0
045   */
046  @Override
047  @Contract("null, _, _ -> fail")
048  default void describeTo(@NotNull Appendable appendable, @Nullable Object value, int maxCount) {
049    describeTo(appendable, value, maxCount, DescriberProperties.MAX_DEPTH);
050  }
051
052  /**
053   * <p>
054   * Add the description of the object to the given appendable.
055   * </p>
056   * <p>
057   * Default implementation using some pre-defined recursion consumer which respects the
058   * given depth.
059   * </p>
060   *
061   * @param appendable appendable to add value's string representation to
062   * @param value      the value to describe
063   * @param maxCount   limits the elements to be added to the appendable; a negative value denotes
064   *                   <em>unlimited</em> - it is recommended to use {@link
065   *                   DescriberProperties#UNLIMITED}
066   * @param maxDepth   the maximum depth to possibly recurse into an object to describe it; a
067   *                   negative value denotes <em>unlimited</em> - it is recommended to use {@link
068   *                   DescriberProperties#UNLIMITED}
069   * @throws NullPointerException            if appendable is {@code null}
070   * @throws DescriberIoException            if a failure occurs accessing the appendable
071   * @throws DescriberNotApplicableException if the value handed over to the Describer cannot
072   *                                         be handled by this describer; you should have called
073   *                                         {@link #test(Object)} before
074   * @see #getRecursiveMeAndOtherConsumer(Appendable, int, int)
075   * @since 1.0.0
076   */
077  @Contract("null, _, _, _ -> fail")
078  default void describeTo(@NotNull Appendable appendable, @Nullable Object value, int maxCount,
079                          int maxDepth) {
080    describeTo(appendable, value, maxCount,
081               getRecursiveMeAndOtherConsumer(appendable, maxDepth, maxCount));
082  }
083
084  /**
085   * <p>
086   * Add the description of the object to the given appendable and recurse into
087   * sub elements of value using the {@code recursiveMeAndOtherConsumer}.
088   * </p>
089   *
090   * @param appendable                  appendable to add value's string representation to
091   * @param value                       the value to describe
092   * @param maxCount                    limits the elements to be added to the appendable; a
093   *                                    negative value denotes
094   *                                    <em>unlimited</em> - it is recommended to use {@link
095   *                                    DescriberProperties#UNLIMITED}
096   * @param recursiveMeAndOtherConsumer consumer for recursion; call it with current value and
097   *                                    sub element as second argument such that the consumer can
098   *                                    detect self-contained object hierarchies
099   * @throws NullPointerException            if appendable and/or recursiveMeAndOtherConsumer is
100   *                                         {@code null}
101   * @throws DescriberIoException            if a failure occurs accessing the appendable
102   * @throws DescriberNotApplicableException if the value handed over to the Describer cannot
103   *                                         be handled by this describer; you should have called
104   *                                         {@link #test(Object)} before
105   * @since 1.0.0
106   */
107  @Contract("null, _, _, _ -> fail; _, _, _, null -> fail")
108  void describeTo(@NotNull Appendable appendable,
109                  @Nullable Object value,
110                  int maxCount,
111                  @NotNull BiConsumer<Object, Object> recursiveMeAndOtherConsumer);
112
113  /**
114   * <p>
115   * Provides a recursive consumer. The first argument to the consumer will be the current
116   * described value, the second one the value to add to the description. These two values
117   * allow to detect self-contained structures and handle them.
118   * </p>
119   * <p>
120   * By default provides a recursive consumer respecting the given depth and detecting
121   * self-contained structures and replace them by a fixed String as given by
122   * {@link DescriberProperties#RECURSION_PLACEHOLDER}.
123   * </p>
124   *
125   * @param appendable appendable to add the string representation to
126   * @param maxDepth   maximum depth
127   * @param maxCount   maximum element count
128   * @return consumer
129   */
130  @NotNull
131  default BiConsumer<Object, Object> getRecursiveMeAndOtherConsumer(
132      @NotNull Appendable appendable,
133      int maxDepth,
134      int maxCount) {
135    return new RecursiveDescriptionConsumer(appendable, maxDepth, maxCount);
136  }
137
138}
