017package com.github.mmichaelis.describeme.library;
018
019import static com.github.mmichaelis.describeme.core.AppendableUtil.silentAppend;
020import static com.google.common.base.Strings.isNullOrEmpty;
021import static java.util.Objects.requireNonNull;
022
023import com.github.mmichaelis.describeme.core.RecursiveDescriber;
024
025import org.jetbrains.annotations.NotNull;
026import org.jetbrains.annotations.Nullable;
027
028import java.util.function.BiConsumer;
029
030/**
031 * Describes throwables with their message and cause.
032 *
033 * @since 1.0.0
034 */
035public class ThrowableDescriber implements RecursiveDescriber {
036
037  @Override
038  public void describeTo(@NotNull Appendable appendable,
039                         @Nullable Object value,
040                         int maxCount,
041                         @NotNull BiConsumer<Object, Object> recursiveMeAndOtherConsumer) {
042    requireNonNull(appendable, "appendable must not be null.");
043    Throwable throwable = (Throwable) validatedValue(value);
044    assert throwable != null : "Validation should have prevented null value.";
045    silentAppend(appendable, throwable.getClass().getName());
046    Throwable cause = throwable.getCause();
047    String throwableMessage = throwable.getMessage();
048    // Quirk: If only cause but no message is given, the throwable message defaults to the
049    // toString() result of the cause especially starts with the cause's class name. As we
050    // want to recurse on our own, this state has to be guessed and message output prevented in
051    // order not to duplicate the message.
052    boolean hasMessage = !isNullOrEmpty(throwableMessage)
053                         && ((cause == null)
054                             || !throwableMessage.startsWith(cause.getClass().getName()));
055    if (hasMessage) {
056      silentAppend(appendable, ": ");
057      silentAppend(appendable, throwableMessage);
058    }
059    if (cause != null) {
060      if (!hasMessage) {
061        silentAppend(appendable, ":");
062      }
063      silentAppend(appendable, " (cause: ");
064      recursiveMeAndOtherConsumer.accept(throwable, cause);
065      silentAppend(appendable, ")");
066    }
067  }
068
069  @Override
070  public boolean test(@Nullable Object value) {
071    return value instanceof Throwable;
072  }
073}
