111   // Idea taken from:
112   // http://stackoverflow.com/questions/23628631/how-to-make-a-lambda-expression-define-tostring-in-java-8
113   @FunctionalInterface
114   private interface ToStringInterface extends Formattable {
115 
116     String asString();
117 
118     @Override
119     default void formatTo(Formatter formatter, int flags, int width, int precision) {
120       String str = asString();
121       //noinspection resource
122       formatter.format(MessageFormat.format("%1${0}.{1}s", width, precision), str);
123       if (str.length() > precision) {
124         try {
125           formatter.out().append("...");
126         } catch (IOException ignored) {
127         }
128       }
129     }
130   }
